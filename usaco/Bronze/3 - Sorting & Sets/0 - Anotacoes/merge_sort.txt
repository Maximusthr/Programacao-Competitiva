    merge_sort(mid + 1, right);
    aux.clear();

    int i = left, j = mid + 1;

    while(i <= mid || j <= right){
        // vocÃª olha os dois arrays dividos lado a lado
        if (j > right || (i <= mid && v[i] < v[j])){
            aux.push_back(v[i]);
            i++;
        }
        else {
            aux.push_back(v[j]);
            j++;
        }
    }

        v[i] = aux[i-left];
    }
